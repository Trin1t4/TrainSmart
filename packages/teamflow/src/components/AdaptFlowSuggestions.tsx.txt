import { useState } from "react";
import { useMutation, useQuery } from "@tanstack/react-query";
import {
  Sparkles,
  AlertTriangle,
  CheckCircle,
  X,
  Loader2,
  ArrowRight,
} from "lucide-react";
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { Button } from "@/components/ui/button";
import { Alert, AlertDescription, AlertTitle } from "@/components/ui/alert";
import { Badge } from "@/components/ui/badge";

interface AdaptFlowSuggestionsProps {
  programId: number;
  autoCheck?: boolean; // Auto-check on mount
}

export default function AdaptFlowSuggestions({
  programId,
  autoCheck = false,
}: AdaptFlowSuggestionsProps) {
  const [selectedSuggestion, setSelectedSuggestion] = useState<number | null>(null);

  // Check if adaptation is needed
  const { data: checkResult, refetch: recheckNeeded } = useQuery({
    queryKey: ["adaptflow-check", programId],
    queryFn: async () => {
      const res = await fetch(`/api/ai/adaptflow/check/${programId}`, {
        credentials: "include",
      });
      if (!res.ok) throw new Error("Failed to check");
      return res.json();
    },
    enabled: autoCheck,
  });

  // Generate suggestions
  const generateMutation = useMutation({
    mutationFn: async (data: {
      triggerType: string;
      triggerData: any;
    }) => {
      const res = await fetch("/api/ai/adaptflow/suggestions", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        credentials: "include",
        body: JSON.stringify({
          programId,
          ...data,
        }),
      });

      if (!res.ok) {
        const error = await res.json();
        throw new Error(error.error || "Failed to generate");
      }

      return res.json();
    },
  });

  // Update action (accept/dismiss)
  const actionMutation = useMutation({
    mutationFn: async ({ id, action }: { id: number; action: string }) => {
      const res = await fetch(`/api/ai/adaptflow/suggestions/${id}/action`, {
        method: "PATCH",
        headers: { "Content-Type": "application/json" },
        credentials: "include",
        body: JSON.stringify({ action }),
      });

      if (!res.ok) throw new Error("Failed to update");
      return res.json();
    },
    onSuccess: () => {
      generateMutation.reset();
      setSelectedSuggestion(null);
    },
  });

  const handleGenerateSuggestions = () => {
    if (checkResult?.needed) {
      generateMutation.mutate({
        triggerType: checkResult.triggerType,
        triggerData: checkResult.triggerData,
      });
    } else {
      // Manual request
      generateMutation.mutate({
        triggerType: "manual_request",
        triggerData: { reason: "Utente ha richiesto suggerimenti" },
      });
    }
  };

  const handleAccept = (suggestionId: number) => {
    actionMutation.mutate({ id: suggestionId, action: "accepted" });
  };

  const handleDismiss = (suggestionId: number) => {
    actionMutation.mutate({ id: suggestionId, action: "dismissed" });
  };

  const getSeverityColor = (priority: string) => {
    switch (priority) {
      case "high":
        return "border-red-500 bg-red-500/10";
      case "medium":
        return "border-amber-500 bg-amber-500/10";
      case "low":
        return "border-blue-500 bg-blue-500/10";
      default:
        return "border-gray-500 bg-gray-500/10";
    }
  };

  return (
    <div className="space-y-4">
      {/* Header */}
      <Card>
        <CardHeader>
          <CardTitle className="flex items-center gap-2">
            <Sparkles className="h-6 w-6 text-primary" />
            AdaptFlow AI - Suggerimenti Intelligenti
          </CardTitle>
          <p className="text-sm text-muted-foreground mt-1">
            L'AI monitora i tuoi workout e suggerisce adattamenti quando necessario
          </p>
        </CardHeader>
        <CardContent>
          {/* Auto-check result */}
          {checkResult?.needed && (
            <Alert className="mb-4 border-amber-500 bg-amber-500/10">
              <AlertTriangle className="h-4 w-4" />
              <AlertTitle>‚ö†Ô∏è Adattamento Consigliato</AlertTitle>
              <AlertDescription>
                {checkResult.triggerType === "pain_detected" &&
                  `Dolore rilevato: ${checkResult.triggerData.painLocation}`}
                {checkResult.triggerType === "fatigue_high" &&
                  `Affaticamento elevato rilevato negli ultimi workout`}
                {checkResult.triggerType === "performance_decline" &&
                  `Calo performance rilevato`}
              </AlertDescription>
            </Alert>
          )}

          <Button
            onClick={handleGenerateSuggestions}
            disabled={generateMutation.isPending}
            size="lg"
            className="w-full"
          >
            {generateMutation.isPending ? (
              <>
                <Loader2 className="mr-2 h-5 w-5 animate-spin" />
                Generazione in corso...
              </>
            ) : (
              <>
                <Sparkles className="mr-2 h-5 w-5" />
                {checkResult?.needed
                  ? "Genera Suggerimenti AI"
                  : "Richiedi Analisi Personalizzata"}
              </>
            )}
          </Button>
        </CardContent>
      </Card>

      {/* Results */}
      {generateMutation.isSuccess && generateMutation.data && (
        <div className="space-y-4">
          {/* Warnings */}
          {generateMutation.data.warnings?.length > 0 && (
            <Alert variant="destructive">
              <AlertTriangle className="h-4 w-4" />
              <AlertTitle>üö® Attenzione</AlertTitle>
              <AlertDescription>
                <ul className="mt-2 space-y-1">
                  {generateMutation.data.warnings.map((warning: string, i: number) => (
                    <li key={i}>‚Ä¢ {warning}</li>
                  ))}
                </ul>
              </AlertDescription>
            </Alert>
          )}

          {/* General Advice */}
          {generateMutation.data.generalAdvice && (
            <Alert className="bg-gradient-to-r from-primary/5 to-primary/10">
              <Sparkles className="h-4 w-4" />
              <AlertTitle>üí° Consiglio Generale</AlertTitle>
              <AlertDescription className="mt-2">
                {generateMutation.data.generalAdvice}
              </AlertDescription>
            </Alert>
          )}

          {/* Alternatives */}
          {generateMutation.data.alternatives?.length > 0 && (
            <Card>
              <CardHeader>
                <CardTitle className="text-lg">
                  üîÑ Esercizi Alternativi Suggeriti
                </CardTitle>
              </CardHeader>
              <CardContent className="space-y-4">
                {generateMutation.data.alternatives.map((alt: any, i: number) => (
                  <Card
                    key={i}
                    className={`${getSeverityColor(alt.priority)} border-2`}
                  >
                    <CardContent className="pt-6">
                      <div className="space-y-3">
                        {/* Priority Badge */}
                        <div className="flex items-center justify-between">
                          <Badge
                            variant={
                              alt.priority === "high"
                                ? "destructive"
                                : alt.priority === "medium"
                                ? "default"
                                : "outline"
                            }
                          >
                            {alt.priority === "high" && "‚ö†Ô∏è Alta Priorit√†"}
                            {alt.priority === "medium" && "üìå Media Priorit√†"}
                            {alt.priority === "low" && "üí° Ottimizzazione"}
                          </Badge>

                          <Badge variant="outline" className="text-xs">
                            Confidence: {Math.round((generateMutation.data.confidence || 0.8) * 100)}%
                          </Badge>
                        </div>

                        {/* From ‚Üí To */}
                        <div className="flex items-center gap-3">
                          <div className="flex-1 p-3 bg-background rounded-lg">
                            <p className="text-xs text-muted-foreground mb-1">
                              Attuale
                            </p>
                            <p className="font-semibold">{alt.from}</p>
                          </div>

                          <ArrowRight className="h-5 w-5 text-muted-foreground" />

                          <div className="flex-1 p-3 bg-background rounded-lg">
                            <p className="text-xs text-muted-foreground mb-1">
                              Alternative
                            </p>
                            <div className="space-y-1">
                              {alt.to.map((exercise: string, j: number) => (
                                <p key={j} className="font-semibold text-sm">
                                  ‚Ä¢ {exercise}
                                </p>
                              ))}
                            </div>
                          </div>
                        </div>

                        {/* Reason */}
                        <div className="p-3 bg-background/50 rounded-lg">
                          <p className="text-sm">
                            <strong>Perch√©:</strong> {alt.reason}
                          </p>
                        </div>

                        {/* Actions */}
                        <div className="flex gap-2 pt-2">
                          <Button
                            onClick={() => handleAccept(generateMutation.data.suggestionId)}
                            disabled={actionMutation.isPending}
                            className="flex-1"
                            size="sm"
                          >
                            <CheckCircle className="mr-2 h-4 w-4" />
                            Accetta
                          </Button>
                          <Button
                            onClick={() => handleDismiss(generateMutation.data.suggestionId)}
                            disabled={actionMutation.isPending}
                            variant="outline"
                            className="flex-1"
                            size="sm"
                          >
                            <X className="mr-2 h-4 w-4" />
                            Ignora
                          </Button>
                        </div>
                      </div>
                    </CardContent>
                  </Card>
                ))}
              </CardContent>
            </Card>
          )}

          {/* No alternatives */}
          {generateMutation.data.alternatives?.length === 0 &&
            generateMutation.data.warnings?.length === 0 && (
              <Alert>
                <CheckCircle className="h-4 w-4 text-green-600" />
                <AlertTitle>‚úÖ Tutto OK!</AlertTitle>
                <AlertDescription>
                  Il tuo programma sembra ottimale. Nessun adattamento necessario al
                  momento.
                </AlertDescription>
              </Alert>
            )}
        </div>
      )}

      {/* Error */}
      {generateMutation.isError && (
        <Alert variant="destructive">
          <AlertTriangle className="h-4 w-4" />
          <AlertTitle>Errore</AlertTitle>
          <AlertDescription>
            {generateMutation.error instanceof Error
              ? generateMutation.error.message
              : "Impossibile generare suggerimenti"}
          </AlertDescription>
        </Alert>
      )}

      {/* Success feedback */}
      {actionMutation.isSuccess && (
        <Alert className="bg-green-500/10 border-green-500">
          <CheckCircle className="h-4 w-4 text-green-600" />
          <AlertTitle>Azione registrata!</AlertTitle>
          <AlertDescription>
            Il tuo feedback √® stato salvato. Continua ad allenarti!
          </AlertDescription>
        </Alert>
      )}
    </div>
  );
}
