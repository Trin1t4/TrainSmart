import { useQuery } from "@tanstack/react-query";
import { Activity, Video, Camera, Sparkles, TrendingUp } from "lucide-react";
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { Progress } from "@/components/ui/progress";
import { Badge } from "@/components/ui/badge";
import { Button } from "@/components/ui/button";

export default function AIUsageStats() {
  const { data: stats, isLoading } = useQuery({
    queryKey: ["ai-usage-stats"],
    queryFn: async () => {
      const res = await fetch("/api/ai/usage-stats", {
        credentials: "include",
      });
      if (!res.ok) throw new Error("Failed to fetch stats");
      return res.json();
    },
  });

  if (isLoading) {
    return (
      <Card>
        <CardContent className="pt-6">
          <div className="animate-pulse space-y-3">
            <div className="h-4 bg-gray-200 rounded w-3/4" />
            <div className="h-4 bg-gray-200 rounded w-1/2" />
          </div>
        </CardContent>
      </Card>
    );
  }

  if (!stats) return null;

  const getTierColor = (tier: string) => {
    switch (tier) {
      case "elite":
        return "from-purple-500 to-pink-500";
      case "premium":
        return "from-blue-500 to-cyan-500";
      case "base":
        return "from-green-500 to-emerald-500";
      default:
        return "from-gray-400 to-gray-500";
    }
  };

  const getTierBadgeVariant = (tier: string) => {
    switch (tier) {
      case "elite":
        return "default";
      case "premium":
        return "secondary";
      case "base":
        return "outline";
      default:
        return "outline";
    }
  };

  const features = [
    {
      name: "Analisi Composizione Corporea",
      icon: Camera,
      used: stats.thisMonth.bodyComposition.used,
      limit: stats.thisMonth.bodyComposition.limit,
      remaining: stats.thisMonth.bodyComposition.remaining,
      color: "text-blue-600",
      bgColor: "bg-blue-500/10",
    },
    {
      name: "Correzione Video Esercizi",
      icon: Video,
      used: stats.thisMonth.formCorrection.used,
      limit: stats.thisMonth.formCorrection.limit,
      remaining: stats.thisMonth.formCorrection.remaining,
      color: "text-purple-600",
      bgColor: "bg-purple-500/10",
    },
    {
      name: "Suggerimenti AdaptFlow",
      icon: Sparkles,
      used: stats.thisMonth.adaptFlow.used,
      limit: stats.thisMonth.adaptFlow.limit,
      remaining: stats.thisMonth.adaptFlow.remaining,
      color: "text-amber-600",
      bgColor: "bg-amber-500/10",
      unlimited: true,
    },
  ];

  return (
    <div className="space-y-6">
      {/* Header Card */}
      <Card className={`bg-gradient-to-r ${getTierColor(stats.tier)} text-white`}>
        <CardHeader>
          <div className="flex items-center justify-between">
            <div>
              <CardTitle className="text-2xl">Piano {stats.tier.toUpperCase()}</CardTitle>
              <p className="text-white/80 mt-1">Le tue feature AI</p>
            </div>
            <Sparkles className="h-12 w-12 opacity-80" />
          </div>
        </CardHeader>
        <CardContent>
          <div className="grid grid-cols-2 gap-4">
            <div>
              <p className="text-white/70 text-sm">Reset mensile</p>
              <p className="text-xl font-bold">
                {new Date(stats.resetDate).toLocaleDateString("it-IT", {
                  day: "numeric",
                  month: "long",
                })}
              </p>
            </div>
            <div>
              <p className="text-white/70 text-sm">Utilizzo totale</p>
              <p className="text-xl font-bold">
                {stats.allTime.bodyComposition + stats.allTime.formCorrection} analisi
              </p>
            </div>
          </div>
        </CardContent>
      </Card>

      {/* Feature Cards */}
      <div className="grid md:grid-cols-3 gap-4">
        {features.map((feature) => {
          const Icon = feature.icon;
          const percentage = feature.unlimited
            ? 0
            : (feature.used / feature.limit) * 100;

          return (
            <Card key={feature.name} className="relative overflow-hidden">
              <div className={`absolute top-0 right-0 w-24 h-24 ${feature.bgColor} rounded-bl-full opacity-50`} />
              
              <CardHeader>
                <div className="flex items-start justify-between">
                  <Icon className={`h-8 w-8 ${feature.color}`} />
                  {feature.unlimited && (
                    <Badge variant="outline" className="text-xs">
                      ILLIMITATO
                    </Badge>
                  )}
                </div>
                <CardTitle className="text-base mt-2">{feature.name}</CardTitle>
              </CardHeader>

              <CardContent>
                {feature.unlimited ? (
                  <div className="text-center py-4">
                    <p className="text-3xl font-bold">âˆž</p>
                    <p className="text-sm text-muted-foreground mt-1">
                      Sempre disponibile
                    </p>
                  </div>
                ) : (
                  <>
                    <div className="flex items-baseline justify-between mb-2">
                      <span className="text-3xl font-bold">{feature.used}</span>
                      <span className="text-muted-foreground">/ {feature.limit}</span>
                    </div>

                    <Progress value={percentage} className="h-2 mb-2" />

                    <p className="text-sm text-muted-foreground">
                      {feature.remaining > 0 ? (
                        <>
                          <span className="font-semibold text-foreground">
                            {feature.remaining}
                          </span>{" "}
                          rimanenti questo mese
                        </>
                      ) : (
                        <span className="text-amber-600 font-semibold">
                          Limite raggiunto
                        </span>
                      )}
                    </p>
                  </>
                )}
              </CardContent>
            </Card>
          );
        })}
      </div>

      {/* Upgrade CTA */}
      {stats.tier !== "elite" && (
        <Card className="bg-gradient-to-r from-primary/5 to-primary/10 border-primary">
          <CardContent className="pt-6">
            <div className="flex items-center justify-between">
              <div className="space-y-1">
                <h3 className="font-bold text-lg flex items-center gap-2">
                  <TrendingUp className="h-5 w-5" />
                  Sblocca piÃ¹ feature AI
                </h3>
                <p className="text-sm text-muted-foreground">
                  {stats.tier === "none" && "Passa a Base per 2 correzioni video/mese"}
                  {stats.tier === "base" && "Passa a Premium per 10 correzioni video/mese"}
                  {stats.tier === "premium" &&
                    "Passa a Elite per analisi illimitate + coach review"}
                </p>
              </div>
              <Button size="lg">
                Upgrade Piano
              </Button>
            </div>
          </CardContent>
        </Card>
      )}

      {/* Usage History Summary */}
      <Card>
        <CardHeader>
          <CardTitle className="text-lg">ðŸ“Š Storico Utilizzo</CardTitle>
        </CardHeader>
        <CardContent>
          <div className="space-y-3">
            <div className="flex items-center justify-between p-3 bg-muted/50 rounded-lg">
              <div className="flex items-center gap-3">
                <Camera className="h-5 w-5 text-blue-600" />
                <span className="font-medium">Analisi Foto Totali</span>
              </div>
              <Badge variant="secondary">{stats.allTime.bodyComposition}</Badge>
            </div>

            <div className="flex items-center justify-between p-3 bg-muted/50 rounded-lg">
              <div className="flex items-center gap-3">
                <Video className="h-5 w-5 text-purple-600" />
                <span className="font-medium">Correzioni Video Totali</span>
              </div>
              <Badge variant="secondary">{stats.allTime.formCorrection}</Badge>
            </div>
          </div>
        </CardContent>
      </Card>
    </div>
  );
}
